```rust
/// Post-filter search results by metadata fields (Option A from docs workflow spec).
/// Applied after vector search, before returning to caller.
pub fn post_filter_metadata(
    results: Vec<SearchResult>,
    content_role: Option<&str>,
    source_origin: Option<&str>,
) -> Vec<SearchResult> {
    if content_role.is_none() && source_origin.is_none() {
        return results;
    }
    results.into_iter().filter(|r| {
        if let Some(role) = content_role {
            if r.entry.metadata.content_role.as_deref() != Some(role) {
                return false;
            }
        }
        if let Some(origin) = source_origin {
            if r.entry.metadata.source_origin.as_deref() != Some(origin) {
                return false;
            }
        }
        true
    }).collect()
}
```

Uses `GraphStore::traverse` for multi-hop BFS traversal and returns neighbor entries. Uses the same `Result<Json<T>, (StatusCode, String)>` error pattern as existing dashboard handlers:

```rust
#[derive(Serialize)]
pub struct NeighborDto {
    pub id: String,
    pub content: String,
    pub relation: String,
    pub direction: String, // "outgoing" or "incoming"
    pub score: Option<f32>,
    pub source_file: Option<String>,
    pub content_role: Option<String>,
    pub source_origin: Option<String>,
}

#[derive(Deserialize)]
pub struct NeighborParams {
    pub depth: Option<usize>,
}

async fn entry_neighbors(
    Path(entry_id): Path<String>,
    Query(params): Query<NeighborParams>,
    State(state): State<Arc<AppState>>,
) -> Result<Json<serde_json::Value>, (StatusCode, String)> {
    let uuid = uuid::Uuid::parse_str(&entry_id)
        .map_err(|_| (StatusCode::BAD_REQUEST, "Invalid UUID".into()))?;
    let depth = params.depth.unwrap_or(1).min(3); // cap at 3 hops per spec

    // Use traverse for multi-hop BFS (not just edges — the depth slider needs this)
    // Use full path for EdgeDirection (same pattern as existing dashboard handlers at line 282)
    let traversed = state.graph.traverse(&uuid, None, corvia_common::types::EdgeDirection::Both, depth).await
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;

    // Also get direct edges to determine relation types and directions
    let edges = state.graph.edges(&uuid, corvia_common::types::EdgeDirection::Both).await
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;

    // Build a map from neighbor_id -> (relation, direction) from direct edges
    let mut edge_info: std::collections::HashMap<uuid::Uuid, (String, String)> = std::collections::HashMap::new();
    for edge in &edges {
        if edge.from == uuid {
            edge_info.insert(edge.to, (edge.relation.clone(), "outgoing".into()));
        } else {
            edge_info.insert(edge.from, (edge.relation.clone(), "incoming".into()));
        }
    }

    let mut neighbors = Vec::new();
    for entry in &traversed {
        if entry.id == uuid { continue; } // skip self
        let (relation, direction) = edge_info
            .get(&entry.id)
            .cloned()
            .unwrap_or(("transitive".into(), "outgoing".into()));
        neighbors.push(NeighborDto {
            id: entry.id.to_string(),
            content: entry.content.chars().take(200).collect(), // preview
            relation,
            direction,
            score: None,
            source_file: entry.metadata.source_file.clone(),
            content_role: entry.metadata.content_role.clone(),
            source_origin: entry.metadata.source_origin.clone(),
        });
    }

    let count = neighbors.len();
    Ok(Json(json!({ "neighbors": neighbors, "count": count })))
}
```

- [ ] **Step 2: Implement `check_misplaced_doc`**

The function takes entries and docs rules, compares `source_file` against expected path patterns based on `content_role` and `source_origin`. If the path matches a `blocked_paths` glob, it's misplaced. Uses the `glob` crate's `Pattern` type (already a dependency of corvia-kernel):

```rust
use corvia_common::config::DocsRulesConfig;

fn check_misplaced_doc(
    entries: &[KnowledgeEntry],
    rules: &DocsRulesConfig,
) -> Vec<Finding> {
    // Pre-compile glob patterns
    let blocked: Vec<glob::Pattern> = rules.blocked_paths.iter()
        .filter_map(|p| glob::Pattern::new(p).ok())
        .collect();

    let mut findings = Vec::new();
    for entry in entries {
        let Some(ref source_file) = entry.metadata.source_file else { continue };
        for pattern in &blocked {
            if pattern.matches(source_file) {
                findings.push(Finding::new(
                    CheckType::MisplacedDoc,
                    &entry.scope_id,
                    vec![entry.id],
                    1.0,
                    format!("File '{}' is in blocked path '{}'", source_file, pattern),
                ));
            }
        }
    }
    findings
}
```

```rust
fn check_coverage_gap(entries: &[KnowledgeEntry]) -> Vec<Finding> {
    let memory_only: Vec<_> = entries.iter()
        .filter(|e| e.metadata.content_role.as_deref() == Some("memory"))
        .collect();
    let has_formal = entries.iter().any(|e| {
        matches!(e.metadata.content_role.as_deref(), Some("design" | "decision" | "plan"))
    });

    if !memory_only.is_empty() && !has_formal {
        vec![Finding::new(
            CheckType::CoverageGap,
            &entries[0].scope_id,
            memory_only.iter().map(|e| e.id).collect(),
            0.7,
            format!(
                "{} memory entries found but no design/decision/plan docs — consider formalizing",
                memory_only.len()
            ),
        )]
    } else {
        Vec::new()
    }
}
```

- [ ] **Step 2: Implement `check_temporal_contradiction`**

Extends the existing `Contradiction` pattern but additionally requires overlapping `valid_from`/`valid_to` windows AND different `source_origin` values. This is an LLM-assisted check (uses embeddings for similarity).

```rust
fn check_temporal_contradiction(
    entries: &[KnowledgeEntry],
    threshold: f32,
) -> Vec<Finding> {
    let mut findings = Vec::new();
    let current: Vec<_> = entries.iter().filter(|e| e.is_current()).collect();

    for i in 0..current.len() {
        for j in (i + 1)..current.len() {
            let a = current[i];
            let b = current[j];

            // Must have different source_origin
            if a.metadata.source_origin == b.metadata.source_origin { continue; }

            // Must both have embeddings for similarity check
            let (Some(ref ea), Some(ref eb)) = (&a.embedding, &b.embedding) else { continue };

            let sim = cosine_similarity(ea, eb);
            if sim >= threshold {
                findings.push(Finding::new(
                    CheckType::TemporalContradiction,
                    &a.scope_id,
                    vec![a.id, b.id],
                    sim,
                    format!(
                        "High similarity ({:.2}) between entries from '{}' and '{}' — possible contradiction",
                        sim,
                        a.metadata.source_origin.as_deref().unwrap_or("unknown"),
                        b.metadata.source_origin.as_deref().unwrap_or("unknown"),
                    ),
                ));
            }
        }
    }
    findings
}
```

```rust
async fn cmd_docs_check(
    check: Option<&str>,
    fix: bool,
    yes: bool,
) -> Result<()> {
    let config = load_config()?;
    let ws = config.workspace.as_ref()
        .ok_or_else(|| anyhow::anyhow!("Not a workspace — 'docs check' requires [workspace] config"))?;
    let docs_rules = ws.docs.as_ref()
        .and_then(|d| d.rules.clone())
        .unwrap_or_default();
    let scope_id = &config.project.scope_id;

    // Determine which checks to run
    let checks: Vec<CheckType> = if let Some(c) = check {
        vec![c.parse().map_err(|e: String| anyhow::anyhow!(e))?]
    } else {
        vec![CheckType::MisplacedDoc, CheckType::CoverageGap, CheckType::TemporalContradiction]
    };

    println!("Running documentation health checks...");

    // Try routing through running server first (same pattern as cmd_reason)
    if let Some(client) = server_client::ServerClient::detect(&config).await {
        println!("(via server at {})\n", client.url());
        // Route each check through the server's /v1/reason endpoint
        for check_type in &checks {
            let response = client.reason(scope_id, Some(check_type.as_str())).await?;
            for (i, finding) in response.findings.iter().enumerate() {
                println!("  [{}] {} (confidence: {:.0}%)", i + 1, finding.check_type, finding.confidence * 100.0);
                println!("      {}", finding.rationale);
            }
        }
        return Ok(());
    }

    // Fallback: run locally (same pattern as cmd_reason lines 1532-1536)
    let (store, graph, _temporal) = connect_full_store(&config).await?;

    // Load entries via knowledge_files::read_scope (NOT dummy embedding search)
    let data_dir = std::path::Path::new(&config.storage.data_dir);
    let entries = corvia_kernel::knowledge_files::read_scope(data_dir, scope_id)?;

    let reasoner = Reasoner::new(&*store, &*graph)
        .with_docs_rules(docs_rules);

    let mut all_findings = Vec::new();
    for check_type in &checks {
        let findings = reasoner.run_check(&entries, scope_id, *check_type).await?;
        all_findings.extend(findings);
    }

    // Display findings
    if all_findings.is_empty() {
        println!("No issues found.");
    } else {
        for (i, finding) in all_findings.iter().enumerate() {
            println!("  {}. [{}] {} (confidence: {:.0}%)",
                i + 1, finding.check_type.as_str(), finding.rationale, finding.confidence * 100.0);
        }

    // Auto-fix mode for MisplacedDoc (skeleton — print plan, don't execute without --yes)

        let misplaced: Vec<_> = all_findings.iter()
            .filter(|f| f.check_type == CheckType::MisplacedDoc)
            .collect();
        if misplaced.is_empty() {
            println!("No misplaced docs to fix.");
            return Ok(());

        if !yes {

            return Ok(());
        }
        // Execute moves — resolve target from docs_rules.repo_docs_pattern
        for finding in misplaced {
            println!("  Moving: {}", finding.rationale);
            // TODO: resolve target path and execute std::fs::rename + git add + git commit
```

- [ ] **Step 2: Implement incremental re-index logic in `cmd_ingest`**

Uses the `ChunkingPipeline` (same as full ingest) to properly chunk files before embedding. Each chunk becomes a separate `KnowledgeEntry` with its own embedding, preserving the same quality as full ingest.

```rust
use corvia_kernel::chunking_pipeline::ChunkingPipeline;
use corvia_kernel::chunking_strategy::SourceMetadata;

async fn cmd_ingest(path: Option<&str>, incremental: bool, files: &[String]) -> Result<()> {
    let config = load_config()?;
    ensure_inference_ready(&config).await?;

    if incremental && !files.is_empty() {
        println!("Incremental re-indexing {} file(s)...", files.len());
        let (store, _graph, temporal) = init_stores(&config).await?;
        let engine = init_engine(&config).await?;
        let pipeline = ChunkingPipeline::with_kernel_defaults(config.chunking.clone());

        let scope_id = &config.project.scope_id;

        for file_path in files {
            let path = std::path::Path::new(file_path);
            if !path.exists() {
                println!("  Skipped (not found): {}", file_path);
                continue;
            }
            println!("  Re-indexing: {}", file_path);

            // 1. Read file and chunk through the standard pipeline
            let content = std::fs::read_to_string(file_path)?;
            let ext = path.extension().and_then(|e| e.to_str()).unwrap_or("");
            let meta = SourceMetadata {
                file_path: file_path.to_string(),
                extension: ext.to_string(),
                language: None, // let pipeline infer
                scope_id: scope_id.clone(),
                source_version: "incremental".into(),
            };
            let (chunks, _relations) = pipeline.process(&content, &meta)?;
            println!("    Chunked into {} pieces", chunks.len());

            // 2. Supersede old entries: find by source_file, set valid_to, link superseded_by
            // This uses the LiteStore's file-based scan (via as_any downcast)
            // to find entries where metadata.source_file matches.
            // For each old entry: old.valid_to = Some(now), old.superseded_by = Some(new_entry.id)
            // Then write updated old entry back to disk.

            // 3. Embed and insert each chunk as a new entry
            for chunk in &chunks {
                let embedding = engine.embed(&chunk.content).await?;
                let mut new_entry = KnowledgeEntry::new(
                    chunk.content.clone(), scope_id.clone(), "incremental".into(),
                );
                new_entry.metadata.source_file = Some(file_path.to_string());

                new_entry.metadata.end_line = Some(chunk.end_line);
                new_entry.embedding = Some(embedding);
                store.insert(&new_entry).await?;
            }
            println!("    Created {} entries (supersedes previous)", chunks.len());
        }
        return Ok(());
    }
```

```rust
//! Hook generation for workspace doc enforcement.
//!
//! Reads [workspace.docs.rules] from corvia.toml and generates:
//! - .corvia/hooks/doc-placement-check.sh (PreToolUse validation)
//! - repos/*/.git/hooks/post-commit (git-triggered incremental re-index)
//! - .claude/settings.json merge (hook registration)

use corvia_common::config::{CorviaConfig, DocsRulesConfig};
use corvia_common::errors::Result;
use std::path::Path;

pub fn generate_hooks(root: &Path, config: &CorviaConfig) -> Result<()> {
    let ws = config.workspace.as_ref()
        .ok_or_else(|| corvia_common::errors::CorviaError::Config("Not a workspace".into()))?;

    let rules = ws.docs.as_ref()
        .and_then(|d| d.rules.as_ref())
        .cloned()
        .unwrap_or_default();

    generate_placement_hook(root, &rules)?;
    generate_post_commit_hooks(root, ws, config)?;
    merge_claude_settings(root)?;

    println!("Generated enforcement hooks from corvia.toml");
    Ok(())
}

fn generate_placement_hook(root: &Path, rules: &DocsRulesConfig) -> Result<()> {
    let hook_dir = root.join(".corvia/hooks");
    std::fs::create_dir_all(&hook_dir)
        .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to create hooks dir: {e}")))?;

    let blocked_cases: String = rules.blocked_paths.iter()
        .map(|p| format!("  {})\n    echo \"BLOCKED: '$FILE_PATH' matches blocked path '{}'.\" >&2\n    exit 2\n    ;;", p, p))
        .collect::<Vec<_>>()
        .join("\n");

    let script = format!(r#"#!/usr/bin/env bash
set -euo pipefail
INPUT=$(cat)
FILE_PATH=$(echo "$INPUT" | jq -r '.file_path // empty')
[ -z "$FILE_PATH" ] && exit 0
case "$FILE_PATH" in *.md|*.mdx|*.rst) ;; *) exit 0 ;; esac
case "$FILE_PATH" in
{blocked_cases}
  repos/*/docs/*|docs/decisions/*|docs/learnings/*|docs/marketing/*|docs/plans/*)
    exit 0 ;;
esac
echo "NOTE: '$FILE_PATH' is in an unusual location for docs." >&2
exit 0
"#);

    let path = hook_dir.join("doc-placement-check.sh");
    std::fs::write(&path, script)
        .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to write hook: {e}")))?;

    #[cfg(unix)]
    {
        use std::os::unix::fs::PermissionsExt;
        std::fs::set_permissions(&path, std::fs::Permissions::from_mode(0o755))
            .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to chmod: {e}")))?;
    }

    println!("  Created: {}", path.display());
    Ok(())
}

fn generate_post_commit_hooks(
    root: &Path,
    ws: &corvia_common::config::WorkspaceConfig,
    _config: &CorviaConfig,
) -> Result<()> {
    for repo in &ws.repos {
        let git_hooks_dir = root.join(&ws.repos_dir).join(&repo.name).join(".git/hooks");
        if !git_hooks_dir.exists() {
            println!("  Skipped: {} (no .git directory)", repo.name);
            continue;
        }

        // Use xargs to safely handle filenames with newlines/spaces
        let hook = r#"#!/usr/bin/env bash
# Generated by: corvia workspace init-hooks
changed=$(git diff-tree --no-commit-id --name-only -r HEAD)
docs_changed=$(echo "$changed" | grep -E '\.(md|toml|yaml|json|rs|py|ts|js)$' || true)
if [ -n "$docs_changed" ]; then
    echo "$docs_changed" | xargs corvia ingest --incremental --files &
fi
"#;

        let path = git_hooks_dir.join("post-commit");
        std::fs::write(&path, hook)
            .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to write hook: {e}")))?;

        #[cfg(unix)]
        {
            use std::os::unix::fs::PermissionsExt;
            std::fs::set_permissions(&path, std::fs::Permissions::from_mode(0o755))
                .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to chmod: {e}")))?;
        }

        println!("  Created: {}", path.display());
    }
    Ok(())
}

/// Merge the PreToolUse Write|Edit hook into .claude/settings.json
fn merge_claude_settings(root: &Path) -> Result<()> {
    let settings_path = root.join(".claude/settings.json");
    let mut settings: serde_json::Value = if settings_path.exists() {
        let content = std::fs::read_to_string(&settings_path)
            .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to read settings: {e}")))?;
        serde_json::from_str(&content)
            .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to parse settings: {e}")))?
    } else {
        serde_json::json!({})
    };

    // Ensure hooks.PreToolUse array exists
    let hooks = settings.as_object_mut().unwrap()
        .entry("hooks").or_insert(serde_json::json!({}));
    let pre_tool = hooks.as_object_mut().unwrap()
        .entry("PreToolUse").or_insert(serde_json::json!([]));
    let arr = pre_tool.as_array_mut().unwrap();

    // Check if Write|Edit hook already registered
    let already_exists = arr.iter().any(|h| {
        h.get("matcher").and_then(|m| m.as_str()) == Some("Write|Edit")
    });

    if !already_exists {
        arr.push(serde_json::json!({
            "matcher": "Write|Edit",
            "hooks": [{"type": "command", "command": "bash .corvia/hooks/doc-placement-check.sh"}]
        }));
        let content = serde_json::to_string_pretty(&settings)
            .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to serialize: {e}")))?;
        std::fs::write(&settings_path, content)
            .map_err(|e| corvia_common::errors::CorviaError::Config(format!("Failed to write: {e}")))?;
        println!("  Updated: {}", settings_path.display());
    } else {
        println!("  Skipped: {} (Write|Edit hook already registered)", settings_path.display());
    }
    Ok(())
}
```
